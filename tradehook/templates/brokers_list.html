    Alpaca order_types:

    stocks: 'simple_order','limit_order','stop_loss_order', 'bracket_order','trailing_stop_order','stop_price_order'

    crypto: 'simple_order', 'limit_order', 'stop_price_order'

    order_type examples:

    simple_order:

    Json:
    {"ticker": "LTCUSD","qty":1, "side": "buy", "type": "market", "time_in_force":"gtc", "order_type": "simple_order", "tradehook_api_key":"2b008cad19d54b0b8a6331a947a974b9", "broker_name": "Alpaca"}

    alert('{"ticker": "' + syminfo.ticker + '", "qty": "1", "side": "buy", "time_in_force": "gtc","broker_name": "Alpaca", "order_type": "simple_order" "tradehook_api_key": "2b008cad19d54b0b8a6331a947a974b9"}',freq=alert.freq_once_per_bar_close)
